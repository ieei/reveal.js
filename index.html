libcubeb

  • github.com/kinetiknz/cubeb
  • Hardware and platform abstraction library
  • C and C++, but exposes a simple C API

cubeb *ctx = NULL;
cubeb_stream *stream = NULL;
cubeb_stream_params params;

cubeb_init(&ctx, "Cubeb audio test");

params.format = CUBEB_SAMPLE_S16NE;
params.rate = 44100;
params.channels = 2;
cubeb_stream_init(ctx, &stream, "test tone", params,
    100, data_cb, state_cb, user_data);
cubeb_stream_start(stream);
...
cubeb_stream_stop(stream);
